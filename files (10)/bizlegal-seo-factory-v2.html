Create seo_pages table

Open Supabase → SQL Editor → paste and run:

CREATE TABLE IF NOT EXISTS seo_pages ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, slug text UNIQUE NOT NULL, title text NOT NULL, meta text, content text, cta_type text DEFAULT 'docstack', page_type text, jurisdiction text, topic text, keywords text[], published boolean DEFAULT false, created_at timestamptz DEFAULT now() ); -- Enable public read access ALTER TABLE seo_pages ENABLE ROW LEVEL SECURITY; CREATE POLICY "public read" ON seo_pages FOR SELECT USING (published = true);
Step 2 — Next.js Route (drop in your project)
2

app/[...slug]/page.tsx

One file handles all 10,000 pages automatically.

// app/[...slug]/page.tsx import { createClient } from '@supabase/supabase-js' import { notFound } from 'next/navigation' import type { Metadata } from 'next' const sb = createClient( process.env.NEXT_PUBLIC_SUPABASE_URL!, process.env.SUPABASE_SERVICE_KEY! ) export async function generateMetadata({ params }): Promise<Metadata> { const slug = params.slug.join('/') const { data } = await sb.from('seo_pages') .select('title,meta').eq('slug', slug).single() return { title: data?.title, description: data?.meta } } export default async function Page({ params }) { const slug = params.slug.join('/') const { data } = await sb.from('seo_pages') .select('*').eq('slug', slug).eq('published', true).single() if (!data) notFound() return ( <main className="max-w-3xl mx-auto px-6 py-16"> <h1 className="text-3xl font-bold mb-6">{data.title}</h1> <div dangerouslySetInnerHTML={{ __html: data.content }} /> <CTABlock type={data.cta_type} /> </main> ) } // CTA targets: docstack=$49 | brai=free scan | tracr=$99 const ctas = { docstack: { h: "Need a lawyer-drafted contract?", btn: "Get Template $49 →", href: "/#docs" }, brai: { h: "Is your project compliant?", btn: "Free Scan →", href: "https://blockchain-agents.onrender.com" }, tracr: { h: "Investigate a wallet or project", btn: "Order Report $99 →", href: "/tracr" } } function CTABlock({ type = 'docstack' }) { const c = ctas[type] ?? ctas.docstack return <div className="mt-16 p-8 bg-gray-50 rounded-xl border"> <h3 className="text-xl font-bold mb-2">{c.h}</h3> <a href={c.href} className="inline-block mt-4 px-6 py-3 bg-black text-white rounded-lg">{c.btn}</a> </div> }
